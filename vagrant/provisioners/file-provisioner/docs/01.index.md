title: File Provisioner

```bash
##
# Git Commands
# 
# @see https://git-scm.com/docs/git-config
#
git config set --file config/.gitconfig user.name "Nurul Amin Muhit"
git config set --file config/.gitconfig user.email "me@muhit.me"
git config set --file config/.gitconfig init.defaultBranch "master"

##
# Vagrant Commands
# 
# @see https://developer.hashicorp.com/vagrant/docs/cli/init
# @see https://developer.hashicorp.com/vagrant/docs/cli/up
#
vagrant init -m hashicorp/bionic64
vagrant up --provider vmware_fusion --provision

vagrant provision file-provisioner
vagrant provision --provision-with git-config
vagrant provision --provision-with git-install

##
# SSH into Vagrant Machine
# Then execute the git-install script inside the machine
#
# @see https://developer.hashicorp.com/vagrant/docs/cli/ssh_config
# @see https://developer.hashicorp.com/vagrant/docs/cli/ssh
# 
vagrant ssh-config
vagrant ssh

sudo chmod +x provision-scripts/git-install.sh
sudo bash -c provision-scripts/git-install.sh

##
# Verify Git Installation and Configs
#
git --version
git config --list --show-origin
```
